116 private void loadMap(String filePath, World world, Tile[][] mapTile, Tile[] textures) {

118 File file = new File(filePath);

120 BufferedReader br = new BufferedReader(fileReader);
121
122 for (int i = 0; i < Const.WRLD_maxRow; i++) {

124
125 for (int j = 0; j < Const.WRLD_maxCol; j++) {
126 String[] numbers = line.split("\\s+");
127 int num = Integer.parseInt(numbers[j]); // Reading the file itself and stocking int read
128
129 Tile tileCurrent = new Tile(textures[num].spriteCntMax, textures[num].spriteSpeed,
130 textures[num].getCollision(), num); // Creating new tile to store with the according texture
131 tileCurrent.setPos(Const.WRLD_tileScreenSize * j, Const.WRLD_tileScreenSize * i);
132
133 mapTile[i][j] = tileCurrent; // Setting the tile to the actual map
134 mapTile[i][j].setTexture(textures[num].image); // Set the current tile texture to what corresponds

136 mapTile[i][j].setCollision(textures[num].getCollision()); // Set the collision factor to the tile

156 public void update(World world) {
157 int playerScreenX = (Const.WDW_width - Const.WRLD_entityScreenSize) / 2;
158 int playerScreenY = (Const.WDW_height - Const.WRLD_entityScreenSize) / 2;
159
160 for (int i = 0; i < Const.WRLD_maxRow; i++) {
161 for (int j = 0; j < Const.WRLD_maxCol; j++) {
162 int worldX = _floorMap[i][j].getPos()[0];
163 int worldY = _floorMap[i][j].getPos()[1];
164
165 // Checks if the player is close enough to the tile to render it to optimize memory and CPU usage
166 if (worldX + Const.WRLD_tileScreenSize > world.player.worldX - playerScreenX
167 && worldX - Const.WRLD_tileScreenSize < world.player.worldX + playerScreenX
168 && worldY + Const.WRLD_tileScreenSize > world.player.worldY - playerScreenY
169 && worldY - Const.WRLD_tileScreenSize < world.player.worldY + playerScreenY) {
170 _floorMap[i][j].updateFrames();
171 _topMap[i][j].updateFrames();
172 }
173 }
