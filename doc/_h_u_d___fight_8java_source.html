57 public void update(){
58 if(Scene.keyH.interactPressed){
59 confirm = true;
60 Scene.keyH.interactPressed = false;
61 }
62
63 if(Scene.keyH.rightPressed || Scene.keyH.leftPressed){
64 switch(selection.toString()){
65 case "ATTACK":
66 selection = Const.Selection.POTION;
67 break;
68
69 case "POTION":
70 selection = Const.Selection.ATTACK;
71 break;
72
73 default: break;
74
75 case "BACK":
76 if(Scene.keyH.leftPressed){
77 selection = Const.Selection.NONE;
78 choice = 1;
79 Scene.keyH.leftPressed = false;
80 }
81 break;
82 }
83
84 }
85 if(page != 0){
86 //List of 4 choices between attacks
93 switch(choice){
94 case 0:
95 if(Scene.keyH.downPressed){
96 choice = 2; break;
97 }
98 if(Scene.keyH.rightPressed){
99 choice = 1; break;
100 }
101 case 1:
102 if(Scene.keyH.downPressed){
103 choice = 3;
104 }
105 if(Scene.keyH.leftPressed){
106 choice = 0;
107 }
108 if(Scene.keyH.rightPressed){
109 selection = Const.Selection.BACK;
110 choice = -1;
111 }
112 break;
113 case 2:
114 if(Scene.keyH.upPressed){
115 choice = 0;
116 }
117 if(Scene.keyH.rightPressed){
118 choice = 3;
119 }
120 break;
121 case 3:
122 if(Scene.keyH.rightPressed){
123 selection = Const.Selection.BACK;
124 choice = -1;
125 }
126 if(Scene.keyH.upPressed){
127 choice = 1;
128 }
129 if(Scene.keyH.leftPressed){
130 choice = 2;
131 }
132 break;
133
134
135 default: break;
136 }
137 }
138
139 //Simply updating colors
140 switch(selection.toString()){
141 case "ATTACK":
142 changeSelectionColor(0, Const.COLOR_BUTTON_text, Const.COLOR_BUTTON_text_highlight, Const.COLOR_BUTTON_bg, Const.COLOR_BUTTON_bg_highlight); break;
143
144 case "POTION":
145 changeSelectionColor(1, Const.COLOR_BUTTON_text, Const.COLOR_BUTTON_text_highlight, Const.COLOR_BUTTON_bg, Const.COLOR_BUTTON_bg_highlight); break;
146
147 case "BACK":
148 changeSelectionColor(2, Const.COLOR_BUTTON_text, Const.COLOR_BUTTON_text_highlight, Const.COLOR_BUTTON_bg, Const.COLOR_BUTTON_bg_highlight); break;
149
150 default:
151 changeSelectionColor(0, Const.COLOR_BUTTON_text, Const.COLOR_BUTTON_text_highlight, Const.COLOR_BUTTON_bg, Const.COLOR_BUTTON_bg_highlight); break;
152 }
153
154 Scene.keyH.rightPressed = false; Scene.keyH.upPressed = false;
155 Scene.keyH.leftPressed = false; Scene.keyH.downPressed = false;
