141 protected void move(World world, int speed, double dt) {
142 if ((dirX == 0 && dirY != 0) || (dirY == 0 && dirX != 0)) {
143 if (dirX == 1) {
144 worldX += speed * dt;
145 }
146 if (dirX == -1) {
147 worldX -= speed * dt;
148 }
149 if (dirY == 1) {
150 worldY += speed * dt;
151 }
152 if (dirY == -1) {
153 worldY -= speed * dt;
154 }
155 }
156 if (dirX != 0 && dirY != 0) {
157 double normSum = Math.sqrt(dirX * dirX + dirY * dirY); //Normalizing vector
158 worldX += (dirX / normSum) * speed * dt;
159 worldY += (dirY / normSum) * speed * dt;
160 }
161 }
