98 public void update() {
99 switch(state){
100 case Const.State.WELCOME:
101 menu.update();
102 break;
103 case Const.State.WORLD:
104 checkPauseScene();
105 HUD_world.update();
106 player.update(dt);
107 tileManager.update(this);
108
109
110 //Update enemy if he's close enough otherwise useless to update (5 tile radius around the screen)
111 for(Enemy enemy : enemies.values()){
112 int playerScreenX = (Const.WDW_width - Const.WRLD_entityScreenSize) / 2;
113 int playerScreenY = (Const.WDW_height - Const.WRLD_entityScreenSize) / 2;
114
115 if(enemy.worldX + 5*Const.WRLD_tileScreenSize > player.worldX - playerScreenX
116 && enemy.worldX - 5*Const.WRLD_tileScreenSize < player.worldX + playerScreenX
117 && enemy.worldY + 5*Const.WRLD_tileScreenSize > player.worldY - playerScreenY
118 && enemy.worldY - 5*Const.WRLD_tileScreenSize < player.worldY + playerScreenY) {
119 enemy.update(dt);
120
121 }
122 }
123 break;
124
125 case Const.State.PAUSE:
126 menu.update();
127 checkPauseScene();
128 break;
129
130 case Const.State.FIGHT:
131 currfight.update(this); break;
132
133 default:
134 break;

162 if(state == Const.State.WORLD){
163 // TILE
164 tileManager.draw(g2, this);

166 for (Props props : objMap.values()) {
167 props.draw(g2, this);
168 }
169 // PLAYER
170 player.drawInWorld(g2, (screenWidth-Const.WRLD_entityScreenSize)/2,
171 (screenHeight-Const.WRLD_entityScreenSize)/2); // Player is always centered to screen
172
173
174

176 for(Enemy enemy : enemies.values()){
177 // Calculate the screen position of the character relative to the player's position
178 int screenX = enemy.worldX - player.worldX + (Const.WDW_width - Const.WRLD_entityScreenSize)/2;
179 int screenY = enemy.worldY - player.worldY + (Const.WDW_height - Const.WRLD_entityScreenSize)/2;
180
181 enemy.drawInWorld(g2, screenX, screenY);
182 }
183 HUD_world.draw(g2);
184
185 }

188 menu.draw(g2);
189 }
