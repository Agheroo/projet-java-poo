90 public void update(){
91 if(Scene.keyH.escPressed || (confirm && selection == Const.Selection.BACK)){
92 if(FightScene.state == Const.FightState.LOST){
93 System.exit(0);
94 }else if(Scene.state == Const.State.WON){
95 System.exit(0);
96 }
97 else{
98 selection = Const.Selection.PLAY;
99 if(page != 0){
100 page--; //Go back to the previous page
101 choiceClass = -1;
102 Scene.keyH.escPressed = false;
103 confirm = false;
104 }
105 }
106 }
107 if(Scene.keyH.downPressed || Scene.keyH.upPressed){
108 if(page == 0){
109 if(selection == Const.Selection.PLAY){
110 changeSelectionColor(1, Const.COLOR_BUTTON_text, Const.COLOR_BUTTON_text_highlight, Const.COLOR_BUTTON_bg, Const.COLOR_BUTTON_bg_highlight);
111 selection = Const.Selection.QUIT;
112 }
113 else if(selection == Const.Selection.QUIT){
114 changeSelectionColor(0, Const.COLOR_BUTTON_text, Const.COLOR_BUTTON_text_highlight, Const.COLOR_BUTTON_bg, Const.COLOR_BUTTON_bg_highlight);
115 selection = Const.Selection.PLAY;
116 }
117 }
118 else{
119 if(Scene.keyH.downPressed && selection == Const.Selection.NONE){
120 selection = Const.Selection.BACK;
121 choiceClass=-1;
122 changeSelectionColor(2, Const.COLOR_BUTTON_text, Const.COLOR_BUTTON_text_highlight, Const.COLOR_BUTTON_bg, Const.COLOR_BUTTON_bg_highlight);
123 }
124 else if(Scene.keyH.upPressed && selection == Const.Selection.BACK){
125 choiceClass = 1;
126 selection = Const.Selection.NONE;
127 changeSelectionColor(0, Const.COLOR_BUTTON_text, Const.COLOR_BUTTON_text_highlight, Const.COLOR_BUTTON_bg, Const.COLOR_BUTTON_bg_highlight);
128 }
129 }
130 Scene.keyH.downPressed = false;
131 Scene.keyH.upPressed = false;
132 }
133 if(Scene.keyH.rightPressed){ //Chose between different character classes
134 choiceClass++;
135 if(choiceClass >= Const.nbOfCharacters)
136 choiceClass = 0;
137 Scene.keyH.rightPressed = false;
138 }
139 if(Scene.keyH.leftPressed){ //Chose between different character classes
140 choiceClass--;
141 if(choiceClass < 0)
142 choiceClass = Const.nbOfCharacters -1;
143 Scene.keyH.leftPressed = false;
144 }
145
146
147
148
149 //Confirming
150 if(Scene.keyH.interactPressed){
151 confirm = true;
152 Scene.keyH.interactPressed = false;
153 }
154
155 //Checking what is confirmed
156 if(confirm){
157 switch(selection.toString()){
158 case "PLAY":
159 page=1;
160 choiceClass = 0;
161 selection = Const.Selection.NONE;
162 break;
163 case "QUIT":
164 System.exit(0); break;
165 case "BACK":
166 page = 0;
167 choiceClass = -1;
168 selection = Const.Selection.PLAY;
169 changeSelectionColor(0, Const.COLOR_BUTTON_text, Const.COLOR_BUTTON_text_highlight, Const.COLOR_BUTTON_bg, Const.COLOR_BUTTON_bg_highlight);
170 break;
171 case "NONE": //Chose the character class and create the game
172 selectClass();
173 Scene.state = Const.State.WORLD;
174 default:break;
175 }
176
177 confirm = false;
178 }
179
180 spriteUpdater++;
181 if (spriteUpdater > 20) {
182 spriteCnt++;
183 if (spriteCnt == Const.nbSpriteCharacterAnim) {
184 spriteCnt = 0;
185 }
186 spriteUpdater = 0;
187 }
